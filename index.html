  import requests
  from bs4 import BeautifulSoup
  # Obtenemos información
  response = requests.get('https://www.filmaffinity.com/es/search.php?stext=tres+metros')
  

  # Procesamos información
  soup = BeautifulSoup(response.content)
  rating_div = soup.find("div", {"class": "avgrat-box"})
  rating_div.get_text()
  '4,8'
